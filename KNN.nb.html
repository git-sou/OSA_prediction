LEARNING PROCESS:

set.seed(123)
train_Control <- trainControl(method="cv", number=10)

k <- expand.grid(k = 2:40) 

knn <- train(IAH ~ Age+Cervical+BMI, data = train,
               method="knn",trControl=train_Control, 
               preProcess="range", tuneGrid = k, metrics ="RMSE")

print(knn)
